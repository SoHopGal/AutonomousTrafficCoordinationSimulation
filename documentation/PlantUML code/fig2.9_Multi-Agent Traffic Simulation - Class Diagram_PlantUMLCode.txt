class Env {
    - vehicles
    - intersections
    - pedestrians
    - light_mode

    + step()
    + spawn_vehicle()
    + run_scenario()
}


class Vehicle {
    - vehicle_id
    - position
    - direction
    - speed
    - is_emergency

    + move()
    + choose_turn()
    + priority()
}


class Intersection {
    - row
    - col
    - faulty

    + step()
    + can_go()
}


class IntersectionLights {
    - horizontal
    - vertical

    + step_fixed()
    + step_adaptive()
    + can_go()
}


class TrafficLight {
    - phase
    - timer

    + step()
    + force_green()
    + force_red()
}


class ReservationManager {
    - slots

    + request()
    + release()
    + force_grant()
}


class Pedestrian {
    - position
    - active

    + step()
    + near_intersection()
}
